# blueprints/auth.py



def _has_permission(role_id, code):

    """检索查角色是否拥有指定权限码（查 role_permissions 表）"""

    if role_id is None:

        return False

    conn = get_db_connection()

    row = conn.execute(

        'SELECT 1 FROM role_permissions WHERE role_id=%s AND permission_code=%s',

        (role_id, code)

    ).fetchone()

    conn.close()

    return row is not None



def require_permission(code):

    """权限校验装饰器，调用前需先经 @login_required 挂载 g.user"""

    def decorator(f):

        @functools.wraps(f)

        def wrapper(*args, **kwargs):

            if not _has_permission(g.user["role_id"], code):

                return jsonify({"status": "error", "msg": "无权限执行此操作"}), 403

            return f(*args, **kwargs)

        return wrapper

    return decorator

# 分词规则优化（V07.5.12）：仅对超过4字的纯中文拆滑动窗口

raw_words = re.split(r'[\s,，；：！？\t\n\r]+', key)

keywords = []

for w in raw_words:

    w = w.strip()

    if len(w) >= 2:

        keywords.append(w)

    # 仅对超过4字的纯中文拆滑动窗口（含英文字母的词保持整体匹配/span>

    if len(w) > 4 and not re.search(r'[a-zA-Z]', w):

        for i in range(len(w) - 1):

            for j in [2, 3]:

                if i + j <= len(w):

                    kw = w[i:i+j]

                    if kw not in keywords:

                        keywords.append(kw)

# blueprints/ai.py ai_parse() 三级修复策略

try:

    parsed = json.loads(ai_text)              # 级：直接解析

except json.JSONDecodeError:

    # 级：补全末尾引号和花括号

    fixed = ai_text.rstrip()

    if fixed.count('"') % 2 != 0:

        fixed += '"'

    open_b = fixed.count('{')

    close_b = fixed.count('}')

    fixed += "}" * (open_b - close_b)

    try:

        parsed = json.loads(fixed)

        op_log(f"AI返回JSON截断已自动修复，补全了{open_b-close_b}个花括号")

    except json.JSONDecodeError:

        # 级：正则提取完整JSON/span>

        m = re.search(r'\{.*\}', ai_text, re.DOTALL)
