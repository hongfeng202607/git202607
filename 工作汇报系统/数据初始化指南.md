# 慧报空间 - 数据初始化指南

## 一、数据模型说明

提交汇报需要哪些数据、按什么顺序创建：

```
① 建部门 → ② 建用户 → ③ 配汇报关系 → ④ 员工写工作记录（草稿或提交）
→ ⑤ 员工提交周期汇报 → ⑥ 领导审批通过 → ⑦ 上级新建汇报时才能看到下属的已通过汇报
```

> **注意：** 工作记录不再使用本地 IndexedDB 存储，所有草稿直接保存在服务端（record_status=0）。

## 二、初始化脚本（Node.js）

将以下内容保存为 `backend/init-demo-data.js`，然后执行：

```bash
cd backend && node init-demo-data.js
```

```javascript
/**
 * 慧报空间 - 演示数据初始化脚本
 * 执行前确保数据库已创建、表结构已初始化
 * 执行：node init-demo-data.js
 */
const { Pool } = require('pg');
const bcrypt = require('bcryptjs');

const pool = new Pool({
  host: '127.0.0.1', port: 5432,
  database: 'work_report_db',
  user: 'postgres', password: '123456'
});

async function main() {
  console.log('开始初始化演示数据...\n');

  // ========== 1. 清理旧数据（按外键顺序） ==========
  await pool.query('DELETE FROM report_record');
  await pool.query('DELETE FROM audit_log');
  await pool.query('DELETE FROM notification');
  await pool.query('DELETE FROM work_record');
  await pool.query('DELETE FROM cycle_report');
  await pool.query('DELETE FROM report_relation');
  await pool.query('DELETE FROM sys_user WHERE username != \'admin\'');
  await pool.query('DELETE FROM department');
  console.log('旧数据已清理\n');

  // ========== 2. 建部门 ==========
  await pool.query(`
    INSERT INTO department (dept_id, dept_name, parent_id, sort_order) VALUES
    (1, '管理层', NULL, 1),
    (2, '技术部', NULL, 2),
    (3, '研发组', 2, 1),
    (4, '测试组', 2, 2)
  `);
  console.log('部门已创建：管理层、技术部、研发组、测试组');

  // ========== 3. 建用户（密码统一：123456） ==========
  const hash = await bcrypt.hash('123456', 10);
  await pool.query(`
    INSERT INTO sys_user (user_id, username, real_name, password, role_type, dept_id) VALUES
    (2, 'zhangzong',  '张总',   $1, 2, 1),   -- 管理者（总监）
    (3, 'wangjingli', '王经理', $1, 2, 2),   -- 管理者（经理）
    (4, 'lizhuguan',  '李主管', $1, 2, 3),   -- 管理者（主管）
    (5, 'zhangsan',   '张三',   $1, 1, 3),   -- 员工
    (6, 'lisi',       '李四',   $1, 1, 3),   -- 员工
    (7, 'wangwu',     '王五',   $1, 1, 4)    -- 员工
  `, [hash]);
  console.log('用户已创建：张总、王经理、李主管、张三、李四、王五（密码均为 123456）');

  // ========== 4. 配汇报关系 ==========
  await pool.query(`
    INSERT INTO report_relation (reporter_id, receiver_id) VALUES
    (3, 2),  -- 王经理 → 张总
    (4, 3),  -- 李主管 → 王经理
    (5, 4),  -- 张三 → 李主管
    (6, 4),  -- 李四 → 李主管
    (7, 3)   -- 王五 → 王经理（王五在王经理下面，不在李主管下面）
  `);
  console.log('汇报关系已创建：张总←王经理←李主管←张三/李四，张总←王经理←王五');

  // ========== 5. 创建演示数据日期 ==========
  const today = new Date();
  const y = today.getFullYear();
  const m = String(today.getMonth() + 1).padStart(2, '0');
  const d = String(today.getDate()).padStart(2, '0');
  const weekStart = `${y}-${m}-${String(1).padStart(2, '0')}`;  // 本月1号作为周期开始
  const weekEnd = `${y}-${m}-${d}`;                               // 今天作为周期结束

  // ========== 6. 员工提交工作记录 ==========
  // 张三的工作记录
  const r1 = await pool.query(
    `INSERT INTO work_record (user_id, record_date, record_content, record_status, submit_time)
     VALUES ($1, $2, $3, 1, NOW()) RETURNING record_id`,
    [5, weekEnd, '完成了用户管理模块的 CRUD 开发\n修复了登录页的样式兼容性问题\n编写了单元测试用例 10 个']
  );
  await pool.query(`UPDATE work_record SET display_id = TO_CHAR(NOW(), 'YYMMDDHH24MI') WHERE record_id = $1`, [r1.rows[0].record_id]);

  // 李四的工作记录
  const r2 = await pool.query(
    `INSERT INTO work_record (user_id, record_date, record_content, record_status, submit_time)
     VALUES ($1, $2, $3, 1, NOW()) RETURNING record_id`,
    [6, weekEnd, '参加了需求评审会议\n完成了数据库表结构设计\n编写了接口文档']
  );
  await pool.query(`UPDATE work_record SET display_id = TO_CHAR(NOW(), 'YYMMDDHH24MI') WHERE record_id = $1`, [r2.rows[0].record_id]);

  console.log('工作记录已创建：张三、李四各提交了 1 条');

  // ========== 7. 员工提交周期汇报 ==========
  const c1 = await pool.query(
    `INSERT INTO cycle_report (user_id, receiver_id, report_type, cycle_start, cycle_end, report_content, generate_type, report_status, submit_time)
     VALUES ($1, $2, 1, $3, $4, $5, 1, 1, NOW()) RETURNING report_id`,
    [5, 4, weekStart, weekEnd, '本周我完成了用户管理模块的开发工作，包括列表查询、新增、编辑、删除功能。\n修复了登录页在不同浏览器下的样式兼容性问题。\n编写了 10 个单元测试用例，覆盖率提升到 85%。']
  );

  const c2 = await pool.query(
    `INSERT INTO cycle_report (user_id, receiver_id, report_type, cycle_start, cycle_end, report_content, generate_type, report_status, submit_time)
     VALUES ($1, $2, 1, $3, $4, $5, 1, 1, NOW()) RETURNING report_id`,
    [6, 4, weekStart, weekEnd, '本周我参与了新项目的需求评审，明确了技术方案。\n完成了数据库的表结构设计，包含 5 张核心表。\n编写了 API 接口文档，覆盖所有核心接口。']
  );

  // 关联工作记录
  await pool.query('INSERT INTO report_record (report_id, record_id) VALUES ($1, $2), ($3, $4)',
    [c1.rows[0].report_id, r1.rows[0].record_id, c2.rows[0].report_id, r2.rows[0].record_id]);

  console.log('周期汇报已创建：张三、李四各提交了 1 篇周报给李主管');

  // ========== 8. 李主管审批通过 ==========
  // 审批张三的汇报
  await pool.query(`
    INSERT INTO audit_log (report_id, operator_id, action_type, action_comment)
    VALUES ($1, 4, 3, '工作完成得不错，继续加油！')
  `, [c1.rows[0].report_id]);
  await pool.query('UPDATE cycle_report SET report_status = 3 WHERE report_id = $1', [c1.rows[0].report_id]);

  // 审批李四的汇报
  await pool.query(`
    INSERT INTO audit_log (report_id, operator_id, action_type, action_comment)
    VALUES ($1, 4, 3, '收到，辛苦了')
  `, [c2.rows[0].report_id]);
  await pool.query('UPDATE cycle_report SET report_status = 3 WHERE report_id = $1', [c2.rows[0].report_id]);

  console.log('审批已通过：李主管审批通过了张三和李四的周报');

  // ========== 9. 发送通知 ==========
  await pool.query(`
    INSERT INTO notification (user_id, type, title, content, report_id) VALUES
    (5, 'approve', '周报已通过', '李主管通过了你的周报，审批意见：工作完成得不错，继续加油！', $1),
    (6, 'approve', '周报已通过', '李主管通过了你的周报，审批意见：收到，辛苦了', $2)
  `, [c1.rows[0].report_id, c2.rows[0].report_id]);
  console.log('通知已发送');

  // ========== 10. 完成 ==========
  console.log('\n==================== 初始化完成 ====================');
  console.log('');
  console.log('登录信息：');
  console.log('  管理员：admin / admin123');
  console.log('  张总：  zhangzong / 123456（可看到所有人的汇报）');
  console.log('  王经理：wangjingli / 123456（可看到李主管、张三、李四的汇报）');
  console.log('  李主管：lizhuguan / 123456（可看到张三、李四的汇报）');
  console.log('  张三：  zhangsan / 123456（普通员工）');
  console.log('  李四：  lisi / 123456（普通员工）');
  console.log('  王五：  wangwu / 123456（普通员工，暂无数据）');
  console.log('');
  console.log('测试流程：');
  console.log('  1. 用李主管登录 → 新建汇报 → 加载记录 → 勾选"包含下属汇报"');
  console.log('     → 出现 "预览" 按钮 → 可查看张三和李四的已通过周报');
  console.log('     → 点 AI 生成 → 汇报内容包含下属的总结');
  console.log('  2. 用王经理登录 → 审批管理 → 可看到李主管及其下属的汇报');

  pool.end();
}

main().catch(err => { console.error('初始化失败:', err); pool.end(); });
```

## 三、手动操作步骤（如果不跑脚本）

如果不想跑脚本，按以下顺序手动操作：

### 第1步：建部门
侧边栏 → 部门管理 → 新建：管理层、技术部、研发组、测试组

### 第2步：建用户
侧边栏 → 用户管理 → 新建用户：
| 用户名 | 姓名 | 角色 | 部门 |
|-------|------|------|------|
| zhangzong | 张总 | 管理者 | 管理层 |
| wangjingli | 王经理 | 管理者 | 技术部 |
| lizhuguan | 李主管 | 管理者 | 研发组 |
| zhangsan | 张三 | 普通员工 | 研发组 |
| lisi | 李四 | 普通员工 | 研发组 |
| wangwu | 王五 | 普通员工 | 测试组 |

密码统一设为 123456

### 第3步：配汇报关系
侧边栏 → 汇报关系 → 依次添加：
- 王经理 → 张总
- 李主管 → 王经理
- 张三 → 李主管
- 李四 → 李主管
- 王五 → 王经理

### 第4步：员工提交记录和汇报
张三登录 → 工作记录 → 提交记录 → 新建汇报（周报）→ 提交
李四登录 → 同上

### 第5步：领导审批
李主管登录 → 审批管理 → 通过张三和李四的周报

### 第6步：测试预览
李主管 → 新建汇报 → 加载记录 → 勾选"包含下属汇报" → 出现预览按钮
